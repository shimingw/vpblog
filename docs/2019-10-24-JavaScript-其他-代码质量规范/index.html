# 定义不确定变量

当定义一个不确定的变量时，使用函数返回这个变量可能的值

//bad
let a;
if (b) {
  a = "11111";
} else {
  a = "22222";
}

//good
function computeA(b) {
  if (b) {
    return "11111";
  } else {
    return "22222";
  }
}
const a = computeA(b);
