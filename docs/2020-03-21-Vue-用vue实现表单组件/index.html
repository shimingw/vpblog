# 组件实现

# index

调用自定的form表单组件

<template>
  <sForm :model="form" ref="form" :rules="rules">
    <formItem label="活动名称" prop="name">
      <inputItem v-model="form.name" />
    </formItem>
    <formItem label="活动地点" prop="area">
      <inputItem v-model="form.area" />
    </formItem>
    <formItem>
      <button @click="submit">提交</button>
    </formItem>
  </sForm>
</template>

<script>
import sForm from "./form";
import formItem from "./formItem";
import inputItem from "./inputItem";

export default {
  components: {
    sForm,
    formItem,
    inputItem
  },
  data() {
    return {
      form: {
        name: "",
        area: ""
      },
      rules: {
        name: [{ required: true, message: "请输入活动名称", trigger: "blur" }],
        area: [{ required: true, message: "请输入活动地点", trigger: "blur" }]
      }
    };
  },
  methods: {
    submit() {
      this.$refs.form.validate(valid => {
        if (valid) {
          console.log("校验通过");
        } else {
          console.log("校验不通过");
        }
      });
    }
  }
};
</script>

# form

  • 接收model和rules并通过provide传递下去
  • validate校验方法是获取到所有需要校验的formItem实例，并触发他们
<template>
  <div>
    <slot></slot>
  </div>
</template>

<script>
export default {
  // 为子组件提供组件实例，方便获取model和rules
  provide() {
    return {
      form: this
    };
  },
  props: {
    // model作用：为了获取表单中所有的值
    model: {
      type: Object,
      require: true
    },
    rules: {
      type: Object
    }
  },
  data() {
    return {};
  },
  methods: {
    validate(cb) {
      //   触发所有的formItem的validate方法
      const tasks = this.$children
        .filter(item => item.prop)
        .map(item => item.validate());

      Promise.all(tasks)
        .then(res => {
          cb(true);
        })
        .catch(err => {
          cb(false);
        });
    }
  }
};
</script>

# formItem

  • 使用async-validator来实现校验功能
  • prop作用：告知formItem，它绑定的是哪个值
    • 方便从rules和model中取值
<template>
  <div>
    <label v-if="label">{{ label }}</label>
    <slot></slot>
    <p v-if="errorMessage">{{ errorMessage }}</p>
  </div>
</template>

<script>
// 提供异步校验功能
import Schema from "async-validator";

export default {
  inject: ["form"],
  props: {
    label: {
      type: String
    },
    // prop作用：告知form表单，这个item所绑定的校验规则
    prop: {
      type: String
    }
  },
  data() {
    return {
      errorMessage: ""
    };
  },
  mounted() {
    // 当inpnut中值发生变化后触发校验
    this.$on("validate", () => {
      this.validate();
    });
  },
  methods: {
    validate() {
      // 获取需要校验的值和对应的校验规则
      const value = this.form.model[this.prop];
      const rules = this.form.rules[this.prop];
      //   使用async-validator来进行校验
      const desc = { [this.prop]: rules };
      const schema = new Schema(desc);
      // return的是校验结果的Promise
      return schema.validate({ [this.prop]: value }, errors => {
        if (errors) {
          this.errorMessage = errors[0].message;
        } else {
          this.errorMessage = "";
        }
      });
    }
  }
};
</script>

# inputItem

  • 自己实现v-model
  • 每次值变化时触发校验
// 父组件引用
<sInput v-model="sInputData" type="text" />

// 子组件实现
<template>
  <div>
    <input ref="input" :value="value" @input="onInput" v-bind="$attrs" />
  </div>
</template>

<script>
export default {
  inheritAttrs: false,
  props: {
    value: {
      type: String,
      default: ""
    }
  },
  methods: {
    onInput(e) {
      this.$emit("input", e.target.value);
      // TODO: 可以优化
      // 触发父组件的校验功能
      this.$parent.$emit("validate");
    }
  }
};
</script>

# 优化

上述input组件中this.$parent.$emit('validate')，可能存在问题。

如果使用时加了其他元素在他们之间，this.$parent就无法获取到formItem

<formItem label="活动地点" prop="area">
    <div>
        <inputItem v-model="form.area" />
    </div>
</formItem>

使用dispatch方法来取代

dispatch会一层一层的向上找，直到找到componentName名称相匹配的组件

  • componentName需要触发的组件实体名称
  • eventName触发的事件
  • params传参
dispatch(componentName, eventName, params) {
    var parent = this.$parent || this.$root
    var name = parent.$options.componentName

    while (parent && (!name || name !== componentName)) {
        parent = parent.$parent

        if (parent) {
            name = parent.$options.componentName
        }
    }
    if (parent) {
        parent.$emit.apply(parent, [eventName].concat(params))
    }
}
