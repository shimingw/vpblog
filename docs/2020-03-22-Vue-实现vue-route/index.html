# 实现vue-router

  • 实现router-link
    • 最终编译成一个具备点击跳转功能的a标签
  • 实现router-view
    • 最终渲染的是路由表里对应的组件
  • 实现路由切换
    • router-link实现路由跳转
    • router-view匹配路由来查找对应的组件进行渲染
    • 路由变化，会触发router-view重新渲染
let Vue

class sRouter {
  // 路由实例化时会传入一个路由数组
  constructor(options) {
    this.$options = options
    this.routeMap = {}
    // 利用了vue的响应机制
    this.app = new Vue({
      data: {
        current: '/'
      }
    })
  }
  init() {
    this.bindEvents()
    this.createRouteMap(this.$options)
    this.initComponent()
  }

  bindEvents() {
    window.addEventListener('load', this.onHashChange.bind(this))
    window.addEventListener('hashchange', this.onHashChange.bind(this))
  }

  onHashChange() {
    this.app.current = window.location.hash.slice(1) || '/'
  }

  createRouteMap(options) {
    options.routes.forEach(option => {
      this.routeMap[option.path] = option.component
    })
  }

  initComponent() {
    //   注册router-link组件
    Vue.component('router-link', {
      // 定义组件接受参数
      props: {
        to: String
      },
      render(h) {
        return <a href={'#' + this.to}>{this.$slots.default}</a>
      }
    })
    //   注册router-view组件
    const _this = this
    Vue.component('router-view', {
      render: h => {
        //   _this.app.current变化后会触发router-view重新render
        const cmp = _this.routeMap[_this.app.current]
        console.log(cmp)
        return h(cmp)
      }
    })
  }
}

sRouter.install = function(vue) {
  // Vue.use(sRouter)时方法被调用
  Vue = vue

  Vue.mixin({
    beforeCreate() {
      if (this.$options.router) {
        // 确保是根组件时执行一次，将router实例放到Vue原型，以后所有组件实例就均有$router
        Vue.prototype.$router = this.$options.router
        this.$options.router.init()
      }
    }
  })
}

export default sRouter

# 优化：实现嵌套router-view

  • 初始化时定义_routerRoot
    • 将本组件实例的_routerRoot指向，父组件实体的_routerRoot
    • 如果是root则，_routerRoot指向自身的_routerRoot
    • 作用是为了递归遍历时，定义递归的终点
  • router-view在render时，在该组件实例的vnode上定义routerView属性为true
    • 为了在递归时能够计算router-view的depth
  • getMatched方法，获取匹配路径的所有组件，以数据形式排列
    • 目的是为了获取嵌套路由的组件实例
    • 根据depth来确定渲染哪层的组件
let Vue

class sRouter {
  // 路由实例化时会传入一个路由数组
  constructor(options) {
    this.$options = options
    this.routeMap = {}
    // 利用了vue的响应机制
    this.app = new Vue({
      data: {
        current: '/'
      }
    })
  }
  init() {
    this.bindEvents()
    this.createRouteMap(this.$options.routes)
    this.initComponent()
  }

  bindEvents() {
    window.addEventListener('load', this.onHashChange.bind(this))
    window.addEventListener('hashchange', this.onHashChange.bind(this))
  }

  onHashChange() {
    this.app.current = window.location.hash.slice(1) || '/'
  }

  createRouteMap(routes, parentPath = '') {
    parentPath = parentPath ? parentPath + '/' : parentPath
    routes.forEach(route => {
      this.routeMap[parentPath + route.path] = route.component
      if (Reflect.has(route, 'children')) {
        this.createRouteMap(route.children, route.path)
      }
    })
  }

  initComponent() {
    //   注册router-link组件
    Vue.component('router-link', {
      // 定义组件接受参数
      props: {
        to: String
      },
      render(h) {
        return <a href={'#' + this.to}>{this.$slots.default}</a>
      }
    })
    //   注册router-view组件
    const _this = this
    Vue.component('router-view', {
      render(h) {
        // 在该组件的vnode上定义该组件含有router-view
        this.$vnode.data.routerView = true
        let depth = 0
        let parent = this.$parent
        // 逐层向上查找router-view嵌套深度
        while (parent._routerRoot !== parent) {
          const vnodeData = parent.$vnode ? parent.$vnode.data : {}
          if (vnodeData.routerView) {
            depth++
          }
          parent = parent.$parent
        }
        // 通过url路径匹配，需要渲染的组件，实现的较为简陋
        const matched = _this.getMatched()
        // 通过嵌套深度获取对应的组件
        const cmp = matched[depth]
        return h(cmp)
      }
    })
  }
  getMatched() {
    const matched = []
    for (const path in this.routeMap) {
      if (this.routeMap.hasOwnProperty(path)) {
        if (this.app.current.includes(path)) {
          const element = this.routeMap[path]
          matched.push(element)
        }
      }
    }
    return matched
  }
}

sRouter.install = function(vue) {
  // Vue.use(sRouter)时方法被调用
  Vue = vue

  Vue.mixin({
    beforeCreate() {
      if (this.$options.router) {
        // 确保是根组件时执行一次，将router实例放到Vue原型，以后所有组件实例就均有$router
        this._routerRoot = this
        Vue.prototype.$router = this.$options.router
        this.$options.router.init()
      } else {
        // 将本组件实例的_routerRoot指向，父组件实体的_routerRoot
        // 如果是root则，_routerRoot指向自身的_routerRoot
        // 作用是为了递归遍历时，定义递归的终点
        this._routerRoot = (this.$parent && this.$parent._routerRoot) || this
      }
    }
  })
}

export default sRouter
