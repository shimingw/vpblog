import React from 'react'
import Tree from './tree'

function index(props) {
  const data = [
    {
      label: '一级 1',
      children: [
        {
          label: '二级 1-1',
          children: [
            {
              label: '三级 1-1-1',
            },
            {
              label: '三级 1-1-2',
            },
          ],
        },
        {
          label: '二级 1-2',
          children: [
            {
              label: '三级 1-2-1',
            },
            {
              label: '三级 1-2-2',
            },
          ],
        },
      ],
    },
  ]

  return <Tree data={data} />
}

export default index

# tree.js

调用递归组件node.js

import React from 'react'
import Node from './node'

function Tree(props) {
  const data = props.data
  return (
    <ul>
      {data.map((node, index) => (
        <Node key={index} data={node} />
      ))}
    </ul>
  )
}

export default Tree

# node.js

递归组件

  • 如果发现数据中children字段是数组，则会继续调用自己
  • 根据数组个数实例化多个自己
import React, { useState } from 'react'

function Node(props) {
  const data = props.data
  const [isExpend, setExpend] = useState(true)

  return (
    <li>
      {/* 渲染本节点 */}
      <h3 onClick={() => setExpend(!isExpend)}>
        {data.children && data.children.length > 0 ? (
          <span>{isExpend ? '-' : '+'}</span>
        ) : null}
        {data.label}
      </h3>
      {/* 根据是否还有子组件，去递归渲染自己的逻辑 */}
      <ul>
        {data.children && data.children.length && isExpend > 0
          ? data.children.map((item, index) => (
              <Node data={item} key={item.label + index} />
            ))
          : null}
      </ul>
    </li>
  )
}

export default Node
