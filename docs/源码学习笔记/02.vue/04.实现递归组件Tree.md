import sTree from './sTree'

### sTree

- 封装递归组件
- 可以对递归组件做一些统一处理

```vue
<template>
  <ul>
    <sTreeNode v-for="(item, index) in data" :data="item" :key="index" />
  </ul>
</template>

<script>
import sTreeNode from './sTreeNode'
export default {
  components: { sTreeNode },
  props: {
    data: {
      type: Array,
      required: true
    }
  },
  data() {
    return {}
  }
}
</script>
```



### sTreeNode

- 通过定义`name`来循环调用自己

```vue
<template>
  <li>
    <h3 @click="toggle">
      <!-- 通过data.children的长度来判断是否显示展开符 -->
      <span v-if="data.children && data.children.length > 0">{{
        isExpend ? '-' : '+'
      }}</span>
      {{ data.label }}
    </h3>
    <!-- 通过是否展开来判断子节点是否展示 -->
    <ul v-if="isExpend">
      <sTreeNode
        v-for="(item, index) in data.children"
        :data="item"
        :key="index"
      />
    </ul>
  </li>
</template>

<script>
export default {
  name: 'sTreeNode',
  props: {
    data: {
      type: Object,
      required: true
    }
  },
  data() {
    return {
      isExpend: true
    }
  },
  methods: {
    toggle() {
      this.isExpend = !this.isExpend
    }
  }
}
</script>
```
