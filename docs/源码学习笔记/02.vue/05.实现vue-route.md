实现一个简易版的`vue-router`

- 实现`router-link`
  - 最终编译成一个具备点击跳转功能的`a`标签
- 实现`router-view`
  - 最终渲染的是路由表里对应的组件
- 实现路由切换
  - `router-link`实现路由跳转
  - `router-view`匹配路由来查找对应的组件进行渲染
  - 路由变化，会触发`router-view`重新渲染

```js
let Vue

class sRouter {
  // 路由实例化时会传入一个路由数组
  constructor(options) {
    this.$options = options
    this.routeMap = {}
    // 利用了vue的响应机制
    this.app = new Vue({
      data: {
        current: '/'
      }
    })
  }
  init() {
    this.bindEvents()
    this.createRouteMap(this.$options)
    this.initComponent()
  }

  bindEvents() {
    window.addEventListener('load', this.onHashChange.bind(this))
    window.addEventListener('hashchange', this.onHashChange.bind(this))
  }

  onHashChange() {
    this.app.current = window.location.hash.slice(1) || '/'
  }

  createRouteMap(options) {
    options.routes.forEach(option => {
      this.routeMap[option.path] = option.component
    })
  }

  initComponent() {
    //   注册router-link组件
    Vue.component('router-link', {
      // 定义组件接受参数
      props: {
        to: String
      },
      render(h) {
        return <a href={'#' + this.to}>{this.$slots.default}</a>
      }
    })
    //   注册router-view组件
    const _this = this
    Vue.component('router-view', {
      render: h => {
        //   _this.app.current变化后会触发router-view重新render
        const cmp = _this.routeMap[_this.app.current]
        console.log(cmp)
        return h(cmp)
      }
    })
  }
}

sRouter.install = function(vue) {
  // Vue.use(sRouter)时方法被调用
  Vue = vue

  Vue.mixin({
    beforeCreate() {
      if (this.$options.router) {
        // 确保是根组件时执行一次，将router实例放到Vue原型，以后所有组件实例就均有$router
        Vue.prototype.$router = this.$options.router
        this.$options.router.init()
      }
    }
  })
}

export default sRouter
```
